 public enum JsonFormatTypes

 	INTEGER,
 	BOOLEAN,
 	OBJECT,
 	ARRAY,
 	NULL,

 	private static final Map<String,JsonFormatTypes> _byLCName = new HashMap<String,JsonFormatTypes>();
